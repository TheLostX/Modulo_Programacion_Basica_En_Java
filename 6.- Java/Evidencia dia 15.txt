    public void switch3(){
        
        Scanner scanner = new Scanner(System.in);
        
        System.out.println("Seleccione una zona \n 1.-America del Norte \n 2.-America Central \n 3.-America del sur"
                + "\n 4.-Europa \n 5.-Asia ");
        
        int opcion = scanner.nextInt();
        float costo = 0f;
        
        System.out.println("Ingrese los gramos");
        float gramos = scanner.nextFloat();
        
        if(gramos>5000.0f){
            System.out.println("Su orden no puede procesarse");
            opcion = 0;
        }
        
        switch (opcion){
            
            case 1:
                costo = gramos *110;
                break;
            
            case 2:
                costo = gramos * 100;
                break;
                
            case 3:
                costo = gramos * 120;
                break;
                
            case 4:
                costo = gramos * 240;
                break;
                
            case 5:
                costo = gramos * 270;
                
            default:
                System.out.println("Orden superior a 5 kg");
                break;
                
        }
        
        System.out.println("El total de su orden es: " +  costo);

    public void numerosParesFor(){
        
        Scanner scanner = new Scanner(System.in);
        
        System.out.println("Ingresa el numero limite");
        
        int limite = scanner.nextInt();
        
        for(int i = 1; i <= limite; i++){
            
            if (i%2==0){
                System.out.println("Numero: " + i);
            }
        }
    }
    
    public void numerosParesWhile(){
        
        Scanner scanner = new Scanner(System.in);
        
        System.out.println("Ingresa el numero limite");
        
        int limite = scanner.nextInt();
        
        int i = 1;
        
        while(i<=limite){
            
            if(i%2==0){
                
                System.out.println("Numero: " + i);
            }
            i++;
        }
