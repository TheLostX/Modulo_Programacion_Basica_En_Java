    public void chismefon(){
        
        Scanner scanner = new Scanner(System.in);
        
        System.out.println("Ingrese el tiempo de llamado");
        int tiempoLlamada = scanner.nextInt();
        float total;
        
        
        if(tiempoLlamada<=5){
            total = tiempoLlamada * 100;
        }
        else if (tiempoLlamada>5 && tiempoLlamada<=8){
            total = (100 * 5) + (tiempoLlamada-5) * 80;
        }
        else if (tiempoLlamada>8 && tiempoLlamada<=10){
            total = (5 * 100) + (3 * 80) + (tiempoLlamada - 8) * 70;
        }
        else{
            total = (5 * 100)+ (3 * 80) + (2 * 70) + (tiempoLlamada - 10) * 50;
        }
        
        System.out.println("ingrese: dia Habil(1), dia domingo(2)");
        byte dia = scanner.nextByte();
        
        if(dia==1){
            System.out.println("ingrese: jornada Vespertina(1), matutina(2)");
            byte jornada = scanner.nextByte();
            
            if(jornada==1){
                float impuesto = total * 0.1f;
                total += impuesto;
            }
            else{
                float impuesto = total * 0.15f;
                total += impuesto;
            }
        }
        else{
            float impuesto = total * 0.03f;
            total += impuesto;
        }
        
        
        System.out.println("El total de la llamada es: " + total);
    }
    
    public void switch1(){
        
        Scanner scanner = new Scanner(System.in);
        
        System.out.println("Ingresa un valor");
        float num1 = scanner.nextFloat();
        
        System.out.println("Ingresa un segundo valor");
        float num2 = scanner.nextFloat();
        
        float resultado = 0;
        
        System.out.println("Ingresa una operacion  \n 1.-Suma \n 2.-Resta \n 3.-Multiplicacion \n 4.- Division");
        
        int operacion = scanner.nextInt();
        
        
        switch (operacion){
            
            case 1:
                resultado = num1 + num2;
                break;
            
            case 2:
                resultado = num1 - num2;
                break;
                
            case 3:
                resultado = num1 * num2;
                break;
                
            case 4:
                resultado = num1 / num2;
                break;
                
            default:
                System.out.println("Opcion no valida");
                break;
                
        }
        
        System.out.println("El resultado es: " + resultado);
    }
