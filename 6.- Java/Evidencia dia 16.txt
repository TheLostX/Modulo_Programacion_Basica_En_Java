    public void promedio(){
        
        Scanner scanner = new Scanner(System.in);
        float suma = 0;
        int cantidad;
        
        do{
            System.out.println("ingresa la cantidad de notas, debe ser al menos 1");
            cantidad = scanner.nextInt();
        }
        while(cantidad <=0);
        
        for(int i = 1; i<=cantidad; i++){
            
            System.out.println("Ingrese la nota " + i);
            float nota = scanner.nextFloat();
            suma += nota;

    public void sumaParMultImpar(){
        
        Scanner scanner = new Scanner(System.in);
        
        
        System.out.println("Ingrese la cantidad de numeros");
        int cantidad = scanner.nextInt();
        int suma = 0;
        int mult = 1;
        int contadorSuma = 0;
        int contadorMult = 0;
        
        if(cantidad<=0){
            mult = 0;
        }
       
        for(int i = 1; i<= cantidad; i++){
            
            System.out.println("Ingrese un numero");
            int numero = scanner.nextInt();
            
            if(numero%2==0){
                suma+=numero; 
                contadorSuma++;
            }
            else{
                mult*=numero;
                contadorMult++;
            }
        }
        
        System.out.println("La suma de los numeros pares es: " + suma);
        System.out.println("La multiplicacion de los numeros impares es: " + mult);
        
        if(contadorMult==contadorSuma){
            
            if (mult == 0){
                System.out.println("No ingresaste ningun numero");
            }
            else{
                System.out.println("Ingresaste la misma cantidad de pares e impares");
            }
        }
        else if(contadorSuma>contadorMult){
            System.out.println("Ingresaste mas numeros pares: Pares = " + contadorSuma + " Impares = " + contadorMult);
        }
        else{
            System.out.println("Ingresaste mas numeros impares: Pares = " + contadorSuma + " Impares = " + contadorMult);
        }
    }
