package com.charles;


public class Libro {
    
    private String tituloLibro;
    private String autor;
    private int numeroEjemplares;
    private int prestamos;
    
    private int maxEjemplares;

    public Libro(String tituloLibro, String autor, int numeroEjemplares, int prestamos) {
        this.tituloLibro = tituloLibro;
        this.autor = autor;
        this.numeroEjemplares = numeroEjemplares;
        this.prestamos = prestamos;
        this.maxEjemplares = numeroEjemplares;
    }
    
    public Libro(){
        tituloLibro = "Vacio";
        autor = "Vacio";
        numeroEjemplares = 0;
        prestamos = 0;
    }

    public String getTituloLibro() {
        return tituloLibro;
    }

    public void setTituloLibro(String tituloLibro) {
        this.tituloLibro = tituloLibro;
    }

    public String getAutor() {
        return autor;
    }

    public void setAutor(String autor) {
        this.autor = autor;
    }

    public int getNumeroEjemplares() {
        return numeroEjemplares;
    }

    public void setNumeroEjemplares(int numeroEjemplares) {
        this.numeroEjemplares = numeroEjemplares;
    }

    public int getPrestamos() {
        return prestamos;
    }

    public void setPrestamos(int prestamos) {
        this.prestamos = prestamos;
    }
    
    public boolean prestamo(int prestamos){
        
        if(numeroEjemplares>0 && prestamos <= numeroEjemplares){
            this.prestamos+=prestamos;
            this.numeroEjemplares-=prestamos;
            System.out.println("Han quedado " + numeroEjemplares + " Libros disponibles");
            System.out.println("Se han prestado : " + this.prestamos + " libros en total");
            return true;
        }
        else{
            System.out.println("Excede la cantidad de ejemplares, solo quedan " + numeroEjemplares + " libros");
            return false;
        }
  
    }
    
    public boolean devolucion(int devolucion){
        
        if(numeroEjemplares==maxEjemplares){
            System.out.println("No hay prestamos realizados");
            return false;
        }
        else if(devolucion>maxEjemplares){
            System.out.println("No se han prestado tantos libros");
            return false;
        }
        else if((devolucion+numeroEjemplares) > maxEjemplares){
            
            System.out.println("Excede la devolucion");
            return false;
        }
  
        else{
            prestamos-=devolucion;
            numeroEjemplares+=devolucion;
            System.out.println("Han quedado " + numeroEjemplares + " Libros disponibles");
            
            return true;
        }
    }
    
    @Override
    public String toString(){
        return "Titulo: " + tituloLibro + "/ Autor: " + autor + "/ Disponibles: " + numeroEjemplares;
    }
    
    
    
}
