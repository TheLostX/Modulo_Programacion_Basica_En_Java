
package com.charles;

import java.util.ArrayList;


public class Usuario {
    
    private String nombre;
    private String nombreUsuario;
    private ArrayList<String> contrasena;
    private static int cantidadUsers = 0;
    
    private int edad;

    public Usuario(String nombre, String nombreUsuario, ArrayList<String> contrasena, int edad) {
        this.nombre = nombre;
        this.nombreUsuario = nombreUsuario;
        this.contrasena = contrasena;
        this.edad = edad;
        cantidadUsers++;
    }

    public String getNombre() {
        return nombre;
    }

    public void setNombre(String nombre) {
        this.nombre = nombre;
    }

    public String getNombreUsuario() {
        return nombreUsuario;
    }

    public void setNombreUsuario(String nombreUsuario) {
        this.nombreUsuario = nombreUsuario;
    }

    public ArrayList<String> getContrasena() {
        return contrasena;
    }

    public void setContrasena(ArrayList<String> contrasena) {
        this.contrasena = contrasena;
    }

    public int getEdad() {
        return edad;
    }

    public void setEdad(int edad) {
        this.edad = edad;
    }
    
    public static int getCantidadUsers(){
        return cantidadUsers;
    }
    
    public String getTodosLosAtributos(){
        
        
       return " Tu nombre es: " + nombre
               + "\n tu nombre de usuario es: " + nombreUsuario
               + "\n tu contraseña es: " + contrasena
               + "\n tu edad es: " + edad;

    }
    

 
}

    public GeneradorContrasena(int capacidad){
        array = new ArrayList<>();
        this.capacidad = capacidad;
    }
        

    public ArrayList generaContrasena(){
        

        //Agrego los caracteres a un arraylist
        for(int i = 0; i < caracteres.length; i++){
            array.add(caracteres[i]);
        }
 
        //Collections.shuffle(array);
        
        //creo un arreglo con capacidad N
        String[] lista = new String[capacidad];
        
        //creo un objeto de la clase Random
        Random random = new Random();
        
        
        //Lleno el arreglo con capacidad N con numeros en posicion N
        for(int i = 0; i < lista.length; i++){

            lista[i] = array.get(random.nextInt(caracteres.length));
        }
